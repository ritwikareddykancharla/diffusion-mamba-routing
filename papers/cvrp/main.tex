\begin{abstract}
We propose \textbf{Diffusion--Mamba}, a hybrid generative optimization
framework for the Capacitated Vehicle Routing Problem (CVRP).
The method couples score-based diffusion priors with a lightweight
Mamba state-space refinement module that enforces capacity and routing
constraints via differentiable projections. The model performs
iterative feasibility-aware refinement and latent-space search to
produce high-quality discrete CVRP solutions. We outline the
architecture, its optimization interpretation, and an evaluation
protocol for benchmarking CVRP solvers.
\end{abstract}

\begin{algorithm}[H]
\caption{Diffusion--Mamba Inference Pipeline}
\label{alg:dm}
\begin{algorithmic}[1]
\STATE Sample initial latent $z_T$
\FOR{$t = T$ down to $1$}
    \STATE $z_{t-1} \leftarrow \text{DDIMStep}(z_t)$
    \STATE $z_{t-1} \leftarrow z_{t-1} + \text{MambaRefine}(z_{t-1})$
    \STATE $z_{t-1} \leftarrow \text{CapacityProject}(z_{t-1})$
\ENDFOR
\STATE Round $z_0$ to discrete CVRP tours
\STATE \textbf{return} best solution
\end{algorithmic}
\end{algorithm}
